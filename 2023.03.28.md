```java
class Solution {

    public boolean PredictTheWinner(int[] nums) {
        int n=nums.length-1;
        int[][] dp=new int [n+1][n+1];
         for (int i = 0; i < n; i++) {
            dp[i][i] = nums[i];
        }
        for(int i=n-1;i>=0;i--){
            for(int j=i+1;j<n+1;j++){
                int right=nums[j]-dp[i][j-1];
                int left=nums[i]-dp[i+1][j];
                dp[i][j]=Math.max(right,left);
            }
        }
        if(dp[0][n]>=0){return true;}
        return false;
    }
}
```
